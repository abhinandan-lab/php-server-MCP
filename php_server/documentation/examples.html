<?php

namespace App\Controllers;

use App\Exceptions\ValidationException;
use App\Exceptions\NotFoundException;

class UserController extends BaseController
{
    /**
     * Get all users with pagination and filtering
     * GET /users?page=1&limit=10&status=active&search=john
     */
    public function getUsers()
    {
        try {
            pp("=== Getting Users List ===");
            
            // Get and validate query parameters
            $page = max(1, (int)($_GET['page'] ?? 1));
            $limit = min(100, max(1, (int)($_GET['limit'] ?? 10)));
            $status = $_GET['status'] ?? null;
            $search = $_GET['search'] ?? null;
            
            pp("Parameters: page=$page, limit=$limit, status=$status, search=$search");
            
            $offset = ($page - 1) * $limit;
            
            // Build dynamic query
            $whereConditions = [];
            $params = [':limit' => $limit, ':offset' => $offset];
            
            if ($status && in_array($status, ['active', 'inactive', 'pending'])) {
                $whereConditions[] = 'status = :status';
                $params[':status'] = $status;
            }
            
            if ($search) {
                $whereConditions[] = '(name LIKE :search OR email LIKE :search)';
                $params[':search'] = '%' . $search . '%';
            }
            
            $whereClause = !empty($whereConditions) ? 'WHERE ' . implode(' AND ', $whereConditions) : '';
            
            // Get users
            $query = "SELECT id, name, email, status, created_at FROM users $whereClause ORDER BY created_at DESC LIMIT :limit OFFSET :offset";
            pp("Executing query: $query");
            
            $users = RunQuery([
                'conn' => $this->conn,
                'query' => $query,
                'params' => $params
            ]);
            
            if (isset($users['error'])) {
                throw new DatabaseException('Failed to retrieve users: ' . $users['error']);
            }
            
            // Get total count for pagination
            $countQuery = "SELECT COUNT(*) as total FROM users $whereClause";
            $countParams = array_filter($params, fn($key) => $key !== ':limit' && $key !== ':offset', ARRAY_FILTER_USE_KEY);
            
            $totalResult = RunQuery([
                'conn' => $this->conn,
                'query' => $countQuery,
                'params' => $countParams
            ]);
            
            $total = (int)($totalResult[0]['total'] ?? 0);
            $totalPages = ceil($total / $limit);
            
            pp("Found $total users, returning page $page of $totalPages");
            
            $this->sendSuccess('Users retrieved successfully', [
                'users' => $users,
                'pagination' => [
                    'page' => $page,
                    'limit' => $limit,
                    'total' => $total,
                    'pages' => $totalPages,
                    'has_next' => $page < $totalPages,
                    'has_prev' => $page > 1
                ],
                'filters' => [
                    'status' => $status,
                    'search' => $search
                ]
            ]);
            
        } catch (\Exception $e) {
            pp("Error in getUsers: " . $e->getMessage());
            throw $e;
        }
    }
    
    /**
     * Create new user
     * POST /users
     */
    public function createUser()
    {
        try {
            pp("=== Creating New User ===");
            
            // Validate input
            $data = $this->validateRequest([
                'name' => 'required|min:2|max:50',
                'email' => 'required|email',
                'password' => 'required|min:8|max:100',
                'role' => 'alpha'
            ]);
            
            pp("Validation passed for user creation");
            ppp($data);
            
            // Check if email already exists
            $existingUser = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT id FROM users WHERE email = :email',
                'params' => [':email' => $data['email']]
            ]);
            
            if (!empty($existingUser)) {
                throw new ValidationException('Email already exists', [
                    'email' => 'This email address is already registered'
                ]);
            }
            
            // Hash password
            $hashedPassword = password_hash($data['password'], PASSWORD_DEFAULT);
            $role = $data['role'] ?? 'user';
            
            // Insert user
            $result = RunQuery([
                'conn' => $this->conn,
                'query' => 'INSERT INTO users (name, email, password_hash, role, status, created_at) VALUES (:name, :email, :password, :role, :status, NOW())',
                'params' => [
                    ':name' => $data['name'],
                    ':email' => $data['email'],
                    ':password' => $hashedPassword,
                    ':role' => $role,
                    ':status' => 'active'
                ],
                'withSuccess' => true
            ]);
            
            if (isset($result['error'])) {
                throw new DatabaseException('Failed to create user: ' . $result['error']);
            }
            
            pp("User created successfully with ID: " . $result['id']);
            
            $this->sendSuccess('User created successfully', [
                'user_id' => $result['id'],
                'name' => $data['name'],
                'email' => $data['email'],
                'role' => $role,
                'status' => 'active'
            ], [
                'Location' => "/api/users/{$result['id']}"
            ]);
            
        } catch (ValidationException $e) {
            pp("Validation error in createUser");
            ppp($e->getValidationErrors());
            throw $e;
        } catch (\Exception $e) {
            pp("Error in createUser: " . $e->getMessage());
            throw $e;
        }
    }
    
    /**
     * Get specific user by ID
     * GET /users/{id}
     */
    public function getUser($id)
    {
        try {
            pp("=== Getting User by ID: $id ===");
            
            if (!is_numeric($id) || $id <= 0) {
                throw new ValidationException('Invalid user ID', [
                    'id' => 'User ID must be a positive integer'
                ]);
            }
            
            $user = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT id, name, email, role, status, created_at, updated_at FROM users WHERE id = :id',
                'params' => [':id' => $id]
            ]);
            
            if (isset($user['error'])) {
                throw new DatabaseException('Failed to retrieve user: ' . $user['error']);
            }
            
            if (empty($user)) {
                throw new NotFoundException('User not found');
            }
            
            pp("User found and retrieved successfully");
            ppp($user[0]);
            
            $this->sendSuccess('User retrieved successfully', $user[0]);
            
        } catch (\Exception $e) {
            pp("Error in getUser: " . $e->getMessage());
            throw $e;
        }
    }
    
    /**
     * Update user information
     * PUT /users/{id}
     */
    public function updateUser($id)
    {
        try {
            pp("=== Updating User ID: $id ===");
            
            if (!is_numeric($id) || $id <= 0) {
                throw new ValidationException('Invalid user ID', [
                    'id' => 'User ID must be a positive integer'
                ]);
            }
            
            // Get current user data
            $currentUser = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT * FROM users WHERE id = :id',
                'params' => [':id' => $id]
            ]);
            
            if (empty($currentUser)) {
                throw new NotFoundException('User not found');
            }
            
            // Get and validate update data
            $data = $this->getRequestData();
            $updateFields = [];
            $params = [':id' => $id];
            
            // Validate and prepare updates
            if (isset($data['name'])) {
                if (strlen($data['name']) < 2 || strlen($data['name']) > 50) {
                    throw new ValidationException('Invalid name', [
                        'name' => 'Name must be between 2 and 50 characters'
                    ]);
                }
                $updateFields[] = 'name = :name';
                $params[':name'] = $data['name'];
            }
            
            if (isset($data['email'])) {
                if (!filter_var($data['email'], FILTER_VALIDATE_EMAIL)) {
                    throw new ValidationException('Invalid email', [
                        'email' => 'Email must be a valid email address'
                    ]);
                }
                
                // Check for email uniqueness
                $emailCheck = RunQuery([
                    'conn' => $this->conn,
                    'query' => 'SELECT id FROM users WHERE email = :email AND id != :id',
                    'params' => [':email' => $data['email'], ':id' => $id]
                ]);
                
                if (!empty($emailCheck)) {
                    throw new ValidationException('Email already exists', [
                        'email' => 'This email address is already registered'
                    ]);
                }
                
                $updateFields[] = 'email = :email';
                $params[':email'] = $data['email'];
            }
            
            if (isset($data['status'])) {
                if (!in_array($data['status'], ['active', 'inactive', 'pending'])) {
                    throw new ValidationException('Invalid status', [
                        'status' => 'Status must be active, inactive, or pending'
                    ]);
                }
                $updateFields[] = 'status = :status';
                $params[':status'] = $data['status'];
            }
            
            if (empty($updateFields)) {
                throw new ValidationException('No valid fields to update', [
                    'fields' => 'At least one field must be provided for update'
                ]);
            }
            
            // Add updated timestamp
            $updateFields[] = 'updated_at = NOW()';
            
            // Execute update
            $query = 'UPDATE users SET ' . implode(', ', $updateFields) . ' WHERE id = :id';
            pp("Executing update query: $query");
            
            $result = RunQuery([
                'conn' => $this->conn,
                'query' => $query,
                'params' => $params,
                'withSuccess' => true
            ]);
            
            if (isset($result['error'])) {
                throw new DatabaseException('Failed to update user: ' . $result['error']);
            }
            
            if ($result['affected_rows'] === 0) {
                pp("No rows affected - user might not exist or no changes made");
                throw new NotFoundException('User not found or no changes made');
            }
            
            // Get updated user data
            $updatedUser = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT id, name, email, role, status, created_at, updated_at FROM users WHERE id = :id',
                'params' => [':id' => $id]
            ]);
            
            pp("User updated successfully");
            ppp($updatedUser[0]);
            
            $this->sendSuccess('User updated successfully', $updatedUser[0]);
            
        } catch (\Exception $e) {
            pp("Error in updateUser: " . $e->getMessage());
            throw $e;
        }
    }
    
    /**
     * Delete user account
     * DELETE /users/{id}
     */
    public function deleteUser($id)
    {
        try {
            pp("=== Deleting User ID: $id ===");
            
            if (!is_numeric($id) || $id <= 0) {
                throw new ValidationException('Invalid user ID', [
                    'id' => 'User ID must be a positive integer'
                ]);
            }
            
            // Check if user exists
            $user = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT id, name, email FROM users WHERE id = :id',
                'params' => [':id' => $id]
            ]);
            
            if (empty($user)) {
                throw new NotFoundException('User not found');
            }
            
            pp("User exists, proceeding with deletion");
            ppp($user[0]);
            
            // Delete user
            $result = RunQuery([
                'conn' => $this->conn,
                'query' => 'DELETE FROM users WHERE id = :id',
                'params' => [':id' => $id],
                'withSuccess' => true
            ]);
            
            if (isset($result['error'])) {
                throw new DatabaseException('Failed to delete user: ' . $result['error']);
            }
            
            if ($result['affected_rows'] === 0) {
                throw new NotFoundException('User not found');
            }
            
            pp("User deleted successfully");
            
            $this->sendSuccess('User deleted successfully', [
                'deleted_user' => $user[0],
                'deleted_at' => date('Y-m-d H:i:s')
            ]);
            
        } catch (\Exception $e) {
            pp("Error in deleteUser: " . $e->getMessage());
            throw $e;
        }
    }
}

-- Blog posts table
CREATE TABLE posts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    slug VARCHAR(200) NOT NULL UNIQUE,
    content TEXT NOT NULL,
    excerpt VARCHAR(500),
    author_id INT NOT NULL,
    category_id INT,
    status ENUM('draft', 'published', 'archived') DEFAULT 'draft',
    featured TINYINT DEFAULT 0,
    published_at DATETIME,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_status (status),
    INDEX idx_author (author_id),
    INDEX idx_published (published_at)
);

-- Categories table
CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    slug VARCHAR(100) NOT NULL UNIQUE,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tags table
CREATE TABLE tags (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    slug VARCHAR(50) NOT NULL UNIQUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Post tags junction table
CREATE TABLE post_tags (
    post_id INT,
    tag_id INT,
    PRIMARY KEY (post_id, tag_id)
);

üéõÔ∏è Blog Controller Example

class PostController extends BaseController
{
    public function createPost()
    {
        try {
            pp("=== Creating New Blog Post ===");
            
            $data = $this->validateRequest([
                'title' => 'required|min:5|max:200',
                'content' => 'required|min:50',
                'excerpt' => 'max:500',
                'category_id' => 'integer',
                'status' => 'alpha',
                'featured' => 'integer'
            ]);
            
            // Generate slug from title
            $slug = $this->generateSlug($data['title']);
            
            // Check slug uniqueness
            $existingPost = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT id FROM posts WHERE slug = :slug',
                'params' => [':slug' => $slug]
            ]);
            
            if (!empty($existingPost)) {
                $slug = $slug . '-' . time();
            }
            
            // Set publish date if publishing
            $publishedAt = null;
            if (($data['status'] ?? 'draft') === 'published') {
                $publishedAt = date('Y-m-d H:i:s');
            }
            
            // Insert post
            $result = RunQuery([
                'conn' => $this->conn,
                'query' => 'INSERT INTO posts (title, slug, content, excerpt, author_id, category_id, status, featured, published_at, created_at) VALUES (:title, :slug, :content, :excerpt, :author_id, :category_id, :status, :featured, :published_at, NOW())',
                'params' => [
                    ':title' => $data['title'],
                    ':slug' => $slug,
                    ':content' => $data['content'],
                    ':excerpt' => $data['excerpt'] ?? null,
                    ':author_id' => 1, // Get from authentication
                    ':category_id' => $data['category_id'] ?? null,
                    ':status' => $data['status'] ?? 'draft',
                    ':featured' => $data['featured'] ?? 0,
                    ':published_at' => $publishedAt
                ],
                'withSuccess' => true
            ]);
            
            if (isset($result['error'])) {
                throw new DatabaseException('Failed to create post: ' . $result['error']);
            }
            
            $postId = $result['id'];
            
            // Handle tags if provided
            if (!empty($data['tags']) && is_array($data['tags'])) {
                $this->attachTagsToPost($postId, $data['tags']);
            }
            
            pp("Blog post created successfully with ID: $postId");
            
            $this->sendSuccess('Post created successfully', [
                'post_id' => $postId,
                'title' => $data['title'],
                'slug' => $slug,
                'status' => $data['status'] ?? 'draft',
                'published_at' => $publishedAt
            ]);
            
        } catch (\Exception $e) {
            pp("Error creating post: " . $e->getMessage());
            throw $e;
        }
    }
    
    private function generateSlug($title)
    {
        $slug = strtolower($title);
        $slug = preg_replace('/[^a-z0-9]+/', '-', $slug);
        $slug = trim($slug, '-');
        return $slug;
    }
    
    private function attachTagsToPost($postId, $tagNames)
    {
        foreach ($tagNames as $tagName) {
            // Create tag if it doesn't exist
            $tagSlug = $this->generateSlug($tagName);
            
            $tag = RunQuery([
                'conn' => $this->conn,
                'query' => 'SELECT id FROM tags WHERE slug = :slug',
                'params' => [':slug' => $tagSlug]
            ]);
            
            if (empty($tag)) {
                $tagResult = RunQuery([
                    'conn' => $this->conn,
                    'query' => 'INSERT INTO tags (name, slug) VALUES (:name, :slug)',
                    'params' => [':name' => $tagName, ':slug' => $tagSlug],
                    'withSuccess' => true
                ]);
                $tagId = $tagResult['id'];
            } else {
                $tagId = $tag[0]['id'];
            }
            
            // Attach tag to post
            RunQuery([
                'conn' => $this->conn,
                'query' => 'INSERT IGNORE INTO post_tags (post_id, tag_id) VALUES (:post_id, :tag_id)',
                'params' => [':post_id' => $postId, ':tag_id' => $tagId]
            ]);
        }
    }
}
